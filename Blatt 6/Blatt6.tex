\teil{Rundungsfehler bei der Gauß-Elimination}

    \[A=\begin{pmatrix}
        8 & 0                   & 8 \\
        4 & \frac{1}{50 \ 000}  & 8 \\
        2 & -2                  & 6 \\
    \end{pmatrix}, b=
    \begin{pmatrix} 4 \\  6 \\ 3 \\ \end{pmatrix}
    \]

    \begin{flalign*}
    a_{11} &= 8 \ \text{pivot} &\\
    l_{21} &= \frac{4}{8} = \frac{1}{2} \ \text{Zeile 2} &\\
    l_{31} &= \frac{2}{8} = \frac{1}{4} \ \text{Zeile 3} &\\[6pt]
    z_2 - \frac{1}{2} z_1 &= (0, \tfrac{1}{50 \ 000}, 4) &\\
    z_3 - \frac{1}{4} z_1 &= (0, -2, 4) &\\[6pt]
    \implies &\begin{pmatrix}
        8 & 0                   & 8 \\

        0 & -2                  & 4 \\
    \end{pmatrix} &\\[6pt]

    l_{32} &= \frac{-2}{\frac{1}{50 \ 000}} = -100 \ 000 &\\
    z_3 - (-100 \ 000)z_2 &= (0, -2, 4) &\\
    &= (0,-2,4) + 100 \ 000 \cdot (0, \tfrac{1}{50 \ 000}, 4) &\\
    &= (0, 0, 400 \ 004) &\\[6pt]
    \implies R &=
    \begin{pmatrix}
        8 & 0                   & 8 \\
        0 & \frac{1}{50 \ 000}  & 4 \\
        0 & -2                  & 4 \\

    \begin{pmatrix}
        1           & 0             & 0 \\

\end{flalign*}
